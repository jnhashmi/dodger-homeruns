<script>

  const TEAM_ID = 119;

  const CURRENT_YEAR = 2026;

  const YEARS = [];

  for (let y = CURRENT_YEAR; y >= 2013; y--) YEARS.push(y);


  const POLL_MS = 60000;

  const MAX_TICKS = 62;

  const BAR_COLOR = '#0977cc';

  const ROW_H = 40;

  const BAR_PADDING = 3;


  const PLACEHOLDER = [

    { name: 'Shohei Ohtani',       hr: 8 },

    { name: 'Freddie Freeman',     hr: 5 },

    { name: 'Teoscar Hernández',   hr: 4 },

    { name: 'Will Smith',          hr: 3 },

    { name: 'Max Muncy',           hr: 3 },

    { name: 'Dalton Rushing',      hr: 2 },

    { name: 'Andy Pages',          hr: 2 },

    { name: 'Hyeseong Kim',        hr: 1 },

    { name: 'Miguel Rojas',        hr: 1 },

    { name: 'Alex Call',           hr: 0 },

    { name: 'Santiago Espinal',    hr: 0 },

    { name: 'Kyle Tucker',         hr: 0 },

    { name: 'Michael Conforto',    hr: 0 },

    { name: 'Tommy Edman',         hr: 0 },

  ];


  let sortMode = 'most';

  let players = [];

  let animating = false;

  let dataLoaded = false;

  let selectedYear = CURRENT_YEAR;

  let pollTimer = null;


  // ── Build year dropdown ──

  const dropdown = document.getElementById('year-dropdown');

  YEARS.forEach(y => {

    const btn = document.createElement('button');

    btn.className = 'year-option' + (y === CURRENT_YEAR ? ' active' : '');

    btn.textContent = y;

    btn.dataset.year = y;

    btn.addEventListener('click', () => selectYear(y));

    dropdown.appendChild(btn);

  });


  document.getElementById('year-btn').addEventListener('click', () => {

    const btn = document.getElementById('year-btn');

    const rect = btn.getBoundingClientRect();

    const dropdown = document.getElementById('year-dropdown');

    dropdown.style.left = rect.left + 'px';

    dropdown.style.top = (rect.bottom + 4) + 'px';

    document.getElementById('year-overlay').classList.add('open');

  });


  document.getElementById('year-overlay').addEventListener('click', (e) => {

    if (e.target === document.getElementById('year-overlay')) {

      document.getElementById('year-overlay').classList.remove('open');

    }

  });


  function selectYear(y) {

    document.getElementById('year-overlay').classList.remove('open');

    if (y === selectedYear) return;

    selectedYear = y;

    document.getElementById('year-btn').textContent = y;

    document.querySelectorAll('.year-option').forEach(b => {

      b.classList.toggle('active', parseInt(b.dataset.year) === y);

    });

    // Stop live polling if not current year

    if (pollTimer) { clearInterval(pollTimer); pollTimer = null; }

    document.getElementById('last-updated').textContent = '';

    document.getElementById('last-updated').style.display = 'none';

    loadYear(y);

  }


  // ── Utilities ──

  function formatTime(d) {

    return d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });

  }


  function sortPlayers(list, mode) {

    const copy = [...list];

    if (mode === 'most') {

      copy.sort((a, b) => b.hr - a.hr || a.name.localeCompare(b.name));

    } else {

      copy.sort((a, b) => a.name.localeCompare(b.name));

    }

    return copy;

  }


  // ── SVG tick builder (draws N filled ticks out of MAX_TICKS slots) ──

  function buildTickSVG(hr, displayHr) {

    // hr = actual count, displayHr = how many to visually show (for animation)

    const d = displayHr !== undefined ? displayHr : hr;

    const slots = Math.max(MAX_TICKS, hr);

    const barH = ROW_H - BAR_PADDING * 2;

    const barY = BAR_PADDING;

    const slotW = 100 / MAX_TICKS;

    const viewW = slots * slotW;

    const gapFrac = 0.18;

    const barW = slotW * (1 - gapFrac);


    const svgNS = 'http://www.w3.org/2000/svg';

    const svg = document.createElementNS(svgNS, 'svg');

    svg.setAttribute('viewBox', `0 0 ${viewW} ${ROW_H}`);

    svg.setAttribute('preserveAspectRatio', 'none');

    svg.style.width = hr > MAX_TICKS ? `${(hr / MAX_TICKS) * 100}%` : '100%';


    for (let i = 0; i < d; i++) {

      const rect = document.createElementNS(svgNS, 'rect');

      rect.setAttribute('x', i * slotW + slotW * (gapFrac / 2));

      rect.setAttribute('y', barY);

      rect.setAttribute('width', barW);

      rect.setAttribute('height', barH);

      rect.setAttribute('fill', BAR_COLOR);

      svg.appendChild(rect);

    }

    return svg;

  }


  // ── Render list (names + 0 ticks initially for grow-in, or full for instant) ──

  function renderList(sorted, instant) {

    const container = document.getElementById('player-list');

    container.innerHTML = '';

    sorted.forEach(p => {

      const row = document.createElement('div');

      row.className = 'player-row';

      row.dataset.name = p.name;

      row.dataset.hr = p.hr;


      const nameEl = document.createElement('div');

      nameEl.className = 'player-name';

      nameEl.textContent = p.name;


      const tickArea = document.createElement('div');

      tickArea.className = 'tick-area';

      tickArea.appendChild(buildTickSVG(p.hr, instant ? p.hr : 0));


      const countEl = document.createElement('div');

      countEl.className = 'hr-count';

      countEl.textContent = instant ? p.hr : 0;


      row.appendChild(nameEl);

      row.appendChild(tickArea);

      row.appendChild(countEl);

      container.appendChild(row);

    });

  }


  // ── HR grow-in animation ──

  // Increments displayed HR count one tick at a time across all rows simultaneously

  function animateHRGrowIn(sorted) {

    const maxHR = Math.max(...sorted.map(p => p.hr), 0);

    if (maxHR === 0) return;


    const rows = document.querySelectorAll('.player-row');

    const rowMap = {};

    rows.forEach(r => { rowMap[r.dataset.name] = r; });


    let tick = 0;

    const TICK_SPEED = 40; // ms per tick — fast progress bar feel


    const timer = setInterval(() => {

      tick++;

      sorted.forEach(p => {

        const row = rowMap[p.name];

        if (!row) return;

        const show = Math.min(tick, p.hr);

        const tickArea = row.querySelector('.tick-area');

        const countEl = row.querySelector('.hr-count');

        tickArea.innerHTML = '';

        tickArea.appendChild(buildTickSVG(p.hr, show));

        countEl.textContent = show;

      });

      if (tick >= maxHR) clearInterval(timer);

    }, TICK_SPEED);

  }


  // ── Sort animation (FLIP) ──

  function animateSort(newSorted) {

    if (animating) return;

    animating = true;


    const container = document.getElementById('player-list');

    const oldPos = {};

    container.querySelectorAll('.player-row').forEach(r => {

      oldPos[r.dataset.name] = r.getBoundingClientRect().top;

    });


    renderList(newSorted, true); // instant, no grow-in on sort


    requestAnimationFrame(() => {

      requestAnimationFrame(() => {

        container.querySelectorAll('.player-row').forEach(r => {

          const old = oldPos[r.dataset.name];

          if (old === undefined) return;

          const diff = old - r.getBoundingClientRect().top;

          if (diff !== 0) {

            r.style.transition = 'none';

            r.style.transform = `translateY(${diff}px)`;

            requestAnimationFrame(() => {

              r.style.transition = 'transform 0.45s cubic-bezier(0.4, 0, 0.2, 1)';

              r.style.transform = '';

            });

          }

        });

        setTimeout(() => { animating = false; }, 500);

      });

    });

  }


  // ── Year transition: fade out → render names → grow in HRs ──

  function transitionToYear(sorted) {

    const container = document.getElementById('player-list');


    // 1. Fade out + slide up

    container.style.transition = 'opacity 0.2s ease, transform 0.2s ease';

    container.style.opacity = '0';

    container.style.transform = 'translateY(-8px)';


    setTimeout(() => {

      // 2. Render names with 0 ticks

      renderList(sorted, false);


      // 3. Fade in

      container.style.transition = 'opacity 0.2s ease, transform 0.2s ease';

      container.style.opacity = '1';

      container.style.transform = 'translateY(0)';


      // 4. After fade in completes, grow in HRs

      setTimeout(() => {

        animateHRGrowIn(sorted);

      }, 220);


    }, 220);

  }


  // ── Fetch data ──

  async function fetchData(year) {

    const url = `https://statsapi.mlb.com/api/v1/teams/${TEAM_ID}/roster?rosterType=active&season=${year}&hydrate=person(stats(type=season,season=${year},group=hitting))`;

    const res = await fetch(url);

    if (!res.ok) throw new Error('API error');

    const data = await res.json();

    return (data.roster || [])

      .filter(p => {

        const pos = p.position?.abbreviation || '';

        const name = p.person?.fullName || '';

        if (pos === 'P' && name !== 'Shohei Ohtani') return false;

        return true;

      })

      .map(p => {

        const stat = p.person?.stats?.[0]?.splits?.[0]?.stat;

        return { name: p.person?.fullName || 'Unknown', hr: stat?.homeRuns ?? 0 };

      });

  }


  // ── Load a year (with transition) ──

  async function loadYear(year, initial) {

    const statusEl = document.getElementById('status-msg');

    const updEl    = document.getElementById('last-updated');


    try {

      const data = await fetchData(year);

      if (data && data.length > 0) {

        players = data;

        dataLoaded = true;

        statusEl.textContent = '';


        if (year === CURRENT_YEAR) {

          updEl.textContent = formatTime(new Date());

          updEl.style.display = '';

        } else {

          updEl.textContent = '';

          updEl.style.display = 'none';

        }


        const sorted = sortPlayers(players, sortMode);


        if (initial) {

          // First load — just render instantly, then grow in

          renderList(sorted, false);

          const container = document.getElementById('player-list');

          container.style.opacity = '1';

          container.style.transform = 'translateY(0)';

          animateHRGrowIn(sorted);

        } else {

          transitionToYear(sorted);

        }


        // Re-enable polling only for current year

        if (year === CURRENT_YEAR) {

          pollTimer = setInterval(() => refreshCurrentYear(), POLL_MS);

        }

      }

    } catch (e) {

      if (!dataLoaded) statusEl.textContent = 'Could not connect. Retrying…';

    }

  }


  async function refreshCurrentYear() {

    if (selectedYear !== CURRENT_YEAR) return;

    try {

      const data = await fetchData(CURRENT_YEAR);

      if (data && data.length > 0) {

        players = data;

        document.getElementById('last-updated').textContent = formatTime(new Date());

        renderList(sortPlayers(players, sortMode), true);

      }

    } catch (e) {}

  }


  // ── Sort button ──

  document.getElementById('sort-btn').addEventListener('click', () => {

    sortMode = sortMode === 'most' ? 'alpha' : 'most';

    animateSort(sortPlayers(players, sortMode));

  });


  // ── Boot ──

  // Show placeholder immediately

  const container = document.getElementById('player-list');

  container.style.opacity = '0';

  players = PLACEHOLDER;

  renderList(sortPlayers(players, sortMode), true);


  // Load real data with grow-in

  loadYear(CURRENT_YEAR, true);

  // Note: poll timer is started inside loadYear on success

</script>
